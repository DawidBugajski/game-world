import { useQueries } from 'react-query';

function GameDetails({ id }) {
  const BASE_URL = 'https://api.rawg.io/api/games';
  const API_KEY = process.env.REACT_APP_RAWG_API_KEY;

  const queries = useQueries([
    {
      queryKey: ['gameDetails', id],
      queryFn: () =>
        axios.get(`${BASE_URL}/${id}?key=${API_KEY}`).then((res) => res.data),
      cacheTime: 10 * 60 * 1000,
      staleTime: 5 * 60 * 1000,
    },
    {
      queryKey: ['gameScreenshots', id],
      queryFn: () =>
        axios
          .get(`${BASE_URL}/${id}/screenshots?key=${API_KEY}`)
          .then((res) => res.data.results),
      cacheTime: 10 * 60 * 1000,
      staleTime: 5 * 60 * 1000,
    },
    {
      queryKey: ['gameTrailers', id],
      queryFn: () =>
        axios
          .get(`${BASE_URL}/${id}/movies?key=${API_KEY}`)
          .then((res) => res.data.results),
      cacheTime: 10 * 60 * 1000,
      staleTime: 5 * 60 * 1000,
    },
    {
      queryKey: ['gameReviews', id],
      queryFn: () =>
        axios
          .get(`${BASE_URL}/${id}/reviews?key=${API_KEY}`)
          .then((res) => res.data.results),
      cacheTime: 10 * 60 * 1000,
      staleTime: 5 * 60 * 1000,
    },
  ]);

  const [gameDetails, gameScreenshots, gameTrailers, gameReviews] = queries.map((query) => query.data);
